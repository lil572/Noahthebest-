// ===== CONFETTI =====
function spawnConfetti(n = 30) {
  const colors = ['#f4a7b9','#d48fa0','#b5c9b0','#d4b8e0','#f0c060','#fce4ec','#e1bee7'];
  for (let i = 0; i < n; i++) {
    const el = document.createElement('div');
    el.className = 'confetti-piece';
    el.style.left = Math.random() * 100 + 'vw';
    el.style.background = colors[Math.floor(Math.random() * colors.length)];
    el.style.width = (8 + Math.random() * 10) + 'px';
    el.style.height = (8 + Math.random() * 10) + 'px';
    el.style.borderRadius = Math.random() > 0.5 ? '50%' : '2px';
    el.style.animationDuration = (2.5 + Math.random() * 3) + 's';
    el.style.animationDelay = (Math.random() * 1) + 's';
    document.body.appendChild(el);
    setTimeout(() => el.remove(), 6000);
  }
}

function spawnHearts() {
  const container = document.getElementById('floatingHearts');
  const hearts = ['💛','🌸','✨','💕','🌷','💫','🎀'];
  for (let i = 0; i < 15; i++) {
    const el = document.createElement('div');
    el.className = 'fheart';
    el.textContent = hearts[Math.floor(Math.random() * hearts.length)];
    el.style.left = Math.random() * 90 + 'vw';
    el.style.bottom = '10px';
    el.style.animationDelay = Math.random() * 2 + 's';
    el.style.fontSize = (1 + Math.random() * 1.5) + 'rem';
    container.appendChild(el);
    setTimeout(() => el.remove(), 5000);
  }
}

// ===== PAGE NAVIGATION =====
function goToPage(n) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.getElementById('page' + n).classList.add('active');
  window.scrollTo(0, 0);
  if (n === 3) loadQuestion();
}

// ===== PAGE 1 - DON'T CLICK =====
let clickCount = 0;
let runPhase = false;
let originalPos = { top: '', left: '', position: '', transform: '' };

function handleClick() {
  const btn = document.getElementById('clickBtn');
  const notice = document.getElementById('runNotice');

  if (runPhase) {
    // Third click — proceed
    notice.textContent = '';
    spawnConfetti(20);
    setTimeout(() => goToPage(2), 500);
    return;
  }

  clickCount++;

  if (clickCount === 1) {
    // First click — move the button randomly
    notice.textContent = 'I said DON\'T! 😤 Try again if you dare...';
    moveButtonRandomly(btn);
  } else if (clickCount === 2) {
    // Second click — move again then return
    notice.textContent = 'Okay FINE. I\'ll let you through... 🙄';
    moveButtonRandomly(btn);
    setTimeout(() => {
      btn.style.position = 'relative';
      btn.style.top = '';
      btn.style.left = '';
      btn.style.transform = '';
      runPhase = true;
      notice.textContent = 'Alright, you actually did it. Come in 💛';
    }, 1200);
  }
}

function moveButtonRandomly(btn) {
  const maxX = 120, maxY = 60;
  const x = (Math.random() * maxX * 2 - maxX);
  const y = (Math.random() * maxY * 2 - maxY);
  btn.style.position = 'relative';
  btn.style.transition = 'all 0.3s ease';
  btn.style.transform = `translate(${x}px, ${y}px)`;
}

function rate(stars) {
  const btns = document.querySelectorAll('.star-btn');
  btns.forEach((b, i) => {
    b.classList.toggle('active', i < stars);
  });

  const data = ratingData[stars];
  const box = document.getElementById('ratingResponse');
  box.style.background = data.color;
  box.innerHTML = `<div class="emoji">${data.emoji}</div><div class="msg">${data.msg}</div>`;

  const nextBtn = document.getElementById('rateNextBtn');
  if (data.showNext) {
    nextBtn.style.display = 'inline-block';
    spawnConfetti(15);
  } else {
    nextBtn.style.display = 'none';
  }
}

// SVG image generators for vision test
function getVisionSVG(type) {
  if (type === 'q1') {
    // Dots pattern hiding number 74
    return `<svg viewBox="0 0 340 180" xmlns="http://www.w3.org/2000/svg" class="vision-svg">
      <rect width="340" height="180" fill="#f5e8ff"/>
      ${generateDotField(340, 180, '#d4b8e0', '#e8d4f8', 12)}
      ${numberDots(74, 170, 90, '#8050a0', '#c090e0', 9)}
    </svg>`;
  } else if (type === 'q2') {
    return `<svg viewBox="0 0 340 180" xmlns="http://www.w3.org/2000/svg" class="vision-svg">
      <rect width="340" height="180" fill="#ffe8f0"/>
      ${generateDotField(340, 180, '#f4c0d0', '#f8d8e8', 11)}
      ${heartDots(170, 95, '#c03060', '#f08090', 9)}
    </svg>`;
  } else {
    return `<svg viewBox="0 0 340 180" xmlns="http://www.w3.org/2000/svg" class="vision-svg">
      <rect width="340" height="180" fill="#e8f5e8"/>
      ${generateDotField(340, 180, '#a0c8a0', '#c8e8c8', 11)}
      ${letterBDots(170, 90, '#204020', '#508050', 9)}
    </svg>`;
  }
}

function generateDotField(w, h, c1, c2, r) {
  let dots = '';
  const colors = [c1, c2];
  for (let i = 0; i < 180; i++) {
    const x = Math.random() * w;
    const y = Math.random() * h;
    const rad = r * (0.6 + Math.random() * 0.8);
    dots += `<circle cx="${x.toFixed(1)}" cy="${y.toFixed(1)}" r="${rad.toFixed(1)}" fill="${colors[i%2]}" opacity="0.7"/>`;
  }
  return dots;
}

function numberDots(num, cx, cy, c1, c2, r) {
  // Draw "74" using dots
  const pattern7 = [
    [0,0],[1,0],[2,0],[2,1],[1,1],[1,2],[1,3],[1,4]
  ];
  const pattern4 = [
    [0,0],[0,1],[0,2],[1,2],[2,2],[2,0],[2,1],[2,3],[2,4]
  ];
  let dots = '';
  const spacing = r * 2.5;
  const offsetX7 = cx - r * 8;
  const offsetY = cy - r * 6;

  for (const [dx, dy] of pattern7) {
    dots += `<circle cx="${(offsetX7 + dx*spacing).toFixed(1)}" cy="${(offsetY + dy*spacing).toFixed(1)}" r="${r}" fill="${c1}" opacity="0.95"/>`;
  }
  const offsetX4 = cx + r * 2;
  for (const [dx, dy] of pattern4) {
    dots += `<circle cx="${(offsetX4 + dx*spacing).toFixed(1)}" cy="${(offsetY + dy*spacing).toFixed(1)}" r="${r}" fill="${c2}" opacity="0.95"/>`;
  }
  return dots;
}

function heartDots(cx, cy, c1, c2, r) {
  let dots = '';
  const points = [];
  for (let t = 0; t <= 2 * Math.PI; t += 0.25) {
    const x = 16 * Math.pow(Math.sin(t), 3);
    const y = -(13 * Math.cos(t) - 5 * Math.cos(2*t) - 2 * Math.cos(3*t) - Math.cos(4*t));
    points.push([cx + x * 2.8, cy + y * 2.8]);
  }
  // Fill interior
  for (let row = -12; row <= 12; row++) {
    for (let col = -12; col <= 12; col++) {
      const x = col * r * 2;
      const y = row * r * 2;
      const t2 = Math.atan2(y, x);
      const hx = 16 * Math.pow(Math.sin(t2), 3) * 2.8;
      const hy = -(13 * Math.cos(t2) - 5 * Math.cos(2*t2) - 2 * Math.cos(3*t2) - Math.cos(4*t2)) * 2.8;
      if (x*x + y*y <= hx*hx + hy*hy) {
        dots += `<circle cx="${(cx + x).toFixed(1)}" cy="${(cy + y).toFixed(1)}" r="${r}" fill="${row%2===0?c1:c2}" opacity="0.95"/>`;
      }
    }
  }
  return dots;
}

function letterBDots(cx, cy, c1, c2, r) {
  const pattern = [
    [0,0],[0,1],[0,2],[0,3],[0,4],
    [1,0],[2,0],[3,0],
    [1,2],[2,2],[3,2],
    [1,4],[2,4],[3,4],
    [4,1],[4,3],
  ];
  let dots = '';
  const spacing = r * 2.5;
  const ox = cx - r * 5, oy = cy - r * 6;
  for (const [dx, dy] of pattern) {
    dots += `<circle cx="${(ox + dx*spacing).toFixed(1)}" cy="${(oy + dy*spacing).toFixed(1)}" r="${r}" fill="${dy%2===0?c1:c2}" opacity="0.95"/>`;
  }
  return dots;
}

let currentQ = 0;
let wrongAnswer = false;

function loadQuestion() {
  const q = questions[currentQ];
  document.getElementById('qProgress').textContent = `Question ${currentQ + 1} of 3`;
  document.getElementById('visionImg').innerHTML = getVisionSVG(q.image);
  document.getElementById('quizFeedback').textContent = q.question;
  document.getElementById('retryBtn').style.display = 'none';
  document.getElementById('quizNextBtn').style.display = 'none';

  const optContainer = document.getElementById('quizOptions');
  optContainer.innerHTML = '';
  q.options.forEach((opt, i) => {
    const btn = document.createElement('button');
    btn.className = 'quiz-opt';
    btn.textContent = opt;
    btn.onclick = () => checkAnswer(i, btn);
    optContainer.appendChild(btn);
  });
}

function checkAnswer(idx, btn) {
  const q = questions[currentQ];
  const allBtns = document.querySelectorAll('.quiz-opt');
  allBtns.forEach(b => b.onclick = null);

  if (idx === q.correct) {
    btn.classList.add('correct');
    document.getElementById('quizFeedback').textContent = '✅ Correct! You passed!';

    if (currentQ < questions.length - 1) {
      setTimeout(() => {
        currentQ++;
        loadQuestion();
      }, 1200);
    } else {
      // All correct!
      spawnConfetti(40);
      spawnHearts();
      document.getElementById('quizFeedback').textContent = '🎉 PERFECT SCORE! You\'re incredible!';
      document.getElementById('quizNextBtn').style.display = 'inline-block';
    }
  } else {
    btn.classList.add('wrong');
    allBtns[q.correct].classList.add('correct');
    document.getElementById('quizFeedback').textContent = '❌ Nope! Back to the start you go...';
    setTimeout(() => {
      document.getElementById('retryBtn').style.display = 'inline-block';
    }, 800);
  }
}

function resetQuiz() {
  currentQ = 0;
  loadQuestion();
}
